20//driver function that initializes the manager terminal
21//param DB - database to be passed to the manager terminal
22pub fn run_man_term(db: &DB) {
23    let mut quit = false;
24    while !quit {
25        let choice = display_options();
26
27        match choice.as_str() {
28            "0" => quit = true,
29            "1" => add_person_ui(db),
30            "2" => remove_person(db),
31            "3" => match db.send_member_reports() {
32                Ok(()) => println!("Member reports sent."),
33                Err(e) => eprintln!("Error sending member reports: {e}"),
34            },
35            "4" => match db.send_provider_reports() {
36                Ok(()) => println!("Provider reports sent."),
37                Err(e) => eprintln!("Error sending provider reports: {e}"),
38            },
39            "5" => match db.send_manager_report() {
40                Ok(()) => println!("Manager report sent."),
41                Err(e) => eprintln!("Error sending manager report: {e}"),
42            },
43            "6" => add_service(db),
44            _ => println!("Invalid input."),
45        }
46    }
47}

49//displays options for the manager terminal
50//returns string
51fn display_options() -> String {
52    println!("----MANAGER TERMINAL----");
53    println!("0. Quit");
54    println!("1. Add new person");
55    println!("2. Remove person");
56    println!("3. Send out member reports");
57    println!("4. Send out provider reports");
58    println!("5. Request manager report");
59    println!("6. Add a service");
60    read_choice()
61}

63//reads the choice of the user as string
64fn read_choice() -> String {
65    use std::io::{self, Write};
66
67    print!("Select an option: ");
68    io::stdout().flush().unwrap();
69
70    let mut buf = String::new();
71    io::stdin().read_line(&mut buf).expect("Invalid input");
72    buf.trim().to_string()
73}

75//adds member to the database
76//param DB - database to add the member too
77fn add_person_ui(db: &DB) {
78    println!("----Add New Person----");
79    let name = read_line("Name: ");
80    let id_str = read_line("9 digit ID: ");
81    let address = read_line("Street address: ");
82    let city = read_line("City name: ");
83    let state = read_line("State (2 letters): ");
84    let zip_str = read_line("5 digit zip: ");
85    let email = read_line("Email: ");
86
87    let id: u32 = match id_str.parse() {
88        Ok(v) => v,
89        Err(_) => {
90            eprintln!("Invalid id, enter valid 9 digit id number");
91            return;
92        }
93    };
94
95    let zipcode: u32 = match zip_str.parse() {
96        Ok(v) => v,
97        Err(_) => {
98            eprintln!("Invalid zip, enter a valid 5 digit number");
99            return;
100        }
101    };
102
103    //set state input to an array of 2 uppercase chars
104    if state.chars().count() != 2 {
105        eprintln!("State must be 2 characters.");
106        return;
107    }
108    let state_upper = state.trim().to_uppercase();
109
110    let location =
111        match LocationInfo::new(&address, &city, &state_upper, zipcode) {
112            Ok(loc) => loc,
113            Err(msg) => {
114                eprintln!("Location error {msg}");
115                return;
116            }
117        };
118
119    let person = match PersonInfo::new(&name, id, &location, &email) {
120        Ok(p) => p,
121        Err(msg) => {
122            eprintln!("Person error {msg}");
123            return;
124        }
125    };
126
127    //determine if user is adding member or provider
128    let person_type = read_line("Is this person a provider? (y/n): ");
129    if person_type.to_lowercase().starts_with('y') {
130        match db.add_provider(&person) {
131            Ok(_) => println!("Provider was successfully added."),
132            Err(e) => eprintln!("Error adding provider: {e}"),
133        }
134    } else {
135        match db.add_member(&person) {
136            Ok(_) => println!("Member was successfully added."),
137            Err(e) => eprintln!("Error adding member: {e}"),
138        }
139    }
140}
141//removes member based off of the member id
142fn remove_person(db: &DB) {
143    let person_type = read_line("Is this person a provider? (y/n)");
144    if person_type.to_lowercase().starts_with('y') {
145        let id_str: String = read_line("Enter the provider ID to be removed: ");
146        let id: u32 = match id_str.trim().parse() {
147            Ok(n) => n,
148            Err(_) => {
149                eprintln!("ID is invalid, please enter valid 9 digit number.");
150                return;
151            }
152        };
153        match db.remove_provider(id) {
154            Ok(()) => println!("Provider removed successfully."),
155            Err(e) => eprintln!("Error removing provider: {e}"),
156        }
157    } else {
158        let id_str: String = read_line("Enter the member ID to remove: ");
159        let id: u32 = match id_str.trim().parse() {
160            Ok(n) => n,
161            Err(_) => {
162                eprintln!("ID is invalid, please enter valid 9 digit number.");
163                return;
164            }
165        };
166        match db.remove_member(id) {
167            Ok(()) => println!("Member removed successfully."),
168            Err(e) => eprintln!("Error removing Member: {e}"),
169        }
170    }
171}

173//helper function to read line from user input
174fn read_line(prompt: &str) -> String {
175    print!("{prompt}");
176    io::stdout().flush().unwrap();
177
178    let mut buf = String::new();
179    io::stdin()
180        .read_line(&mut buf)
181        .expect("Failed to read line");
182    buf.trim().to_string()
183}
